3 # A simple script for running a Lennard-Jones simulation.
4 
5 # Statistics such as energy, (instantaneous) temperature, and (instantaneous) pressure are
6 # output to "LJ-example.npz", and coordinates are output to "LJ-example.xyz".
7 # These xyz files can be read by VMD; a file named "LJ-example.tcl" is also created for use with vmd.
8 # You can then run 'vmd -e LJ-example.tcl LJ-example.xyz' to see the simulation.
9 
10 # This was written for Python 3.4+; it may take some small effort to use a lower version of Python.

13 from math import pi, sqrt
14 import numpy as np
15 import pyparm.d3 as sim # or d2 for 2D simulation
16 import pyparm.util as util

23 tcl_file = 'LJ-example.tcl'
24 xyz_file = 'LJ-example.xyz'

26 # system parameters
27 N = 40
28 N1, N2 = (N//2), (N - (N//2))
29 T0 = 1
30 phi = 0.3 # "packing fraction", i.e. (volume of spheres) / (volume of box)
31 sigmas = ([1] * N1) + ([1.4] * N2)
32 
33 # simulation parameters
34 dt = 0.001
35 total_time = 100
36 data_n = 1000
37 xyz_n = 1000
38 print_n = 20
39 
40 total_steps = int(total_time / dt + 0.5)
41 xyz_dt = float(total_time) / xyz_n
42 data_dt = float(total_time) / data_n
43 print_dt = float(total_time) / print_n
44 
45 # purely calculated
46 d = sim.NDIM # number of dimensions
47 volumes = [s**d * pi/(2*d) for s in sigmas]
48 masses = [v / min(volumes) for v in volumes] # mass proportional to volume, but 1 for the smallest
49 L = (sum(volumes) / phi) ** (1.0/d) # length of each side of the box
50 
51 #---------------------------------------------------------------------------------------------------
52 # Setting up the simulation
53 box = sim.OriginBox(L)
54 atoms = sim.AtomVec(masses)
55 neighbors = sim.NeighborList(box, atoms, 0.4) # the NeighborList, for keeping track of what atoms are near what other atoms
56 LJ = sim.LJRepulse(atoms, neighbors)
57 collec = sim.CollectionVerlet(box, atoms, dt, [LJ], [neighbors]) # the integrator
58 # We use a simple velocity-verlet integrator, which is time-reversible and NVE ensemble
59 # i.e., it preserves number of atoms, volume of box, and energy
60 
61 #---------------------------------------------------------------------------------------------------
62 # Initial Conditions
63 # Now we have created our Atom, but we need to add our atoms to it. We do that in a way that prevents overlap
64 
65 E0 = 0
66 for a,s in zip(atoms, sigmas):
67  E = E0 + 10
68  a.v = sim.rand_vec() # from a gaussian distribution
69  LJ.add(sim.EpsSigAtom(a, 1, s))
70  while E > E0 + 0.1:
71  a.x = box.rand_loc()
72  neighbors.update_list()
73  E = LJ.energy(box)
74  E0 = E
75 
76 collec.reset_com_velocity() # subtract center-of-mass velocity from all particles
77 collec.scale_velocities_to_temp(T0) # scale all velocities to get an instantaneous temperature T = T0, at least at the beginning

99  delim = '\t'
